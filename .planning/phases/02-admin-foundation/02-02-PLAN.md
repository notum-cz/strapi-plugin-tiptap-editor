  **Step 1: Create admin/src/components/FeatureGuard.tsx:**

  Per user's locked decision in CONTEXT.md: "Use a generic FeatureGuard component that conditionally renders children. Hooks are never called for disabled features — FeatureGuard prevents the component from mounting."

  ```typescript
  import { ReactNode } from 'react';
  import { TiptapPresetConfig, isFeatureEnabled } from '../../../shared/types';

  interface FeatureGuardProps {
    /** The feature config value from TiptapPresetConfig (e.g., config.bold, config.heading) */
    featureValue: TiptapPresetConfig[keyof TiptapPresetConfig];
    children: ReactNode;
  }

  /**
   * Conditionally renders children based on whether a feature is enabled in the preset config.
   * When disabled, returns null — children never mount, so any hooks inside are never called.
   * This is the primary mechanism for HOOKS-01/HOOKS-02: hooks don't need internal config gating
   * because FeatureGuard prevents them from being called entirely.
   */
  export function FeatureGuard({ featureValue, children }: FeatureGuardProps): ReactNode {
    if (!isFeatureEnabled(featureValue)) {
      return null;
    }
    return children;
  }
  ```

  This is intentionally simple — a thin wrapper around `isFeatureEnabled`. The value is in the PATTERN it establishes for Phase 3 wiring: each toolbar section in RichTextInput will be wrapped in `<FeatureGuard featureValue={config.bold}>...</FeatureGuard>`, and when the feature is disabled, the entire subtree (including hooks) never mounts.

  Per CONTEXT.md decisions:
  - Error boundary wraps `BaseTiptapInput` only (not the Field.Root/Label/Hint/Error)
  - Fallback UI: "Editor failed to load" message with a retry button that remounts the component
  - Use Strapi design-system components for consistent styling
  - Log errors via `console.error` only — no server-side logging

  Implementation:
  ```typescript
  import React, { Component, ErrorInfo, ReactNode } from 'react';
  import { Box, Typography, Button } from '@strapi/design-system';

  interface Props {
    children: ReactNode;
  }

  interface State {
    hasError: boolean;
    error: Error | null;
  }

  export class EditorErrorBoundary extends Component<Props, State> {
    state: State = { hasError: false, error: null };

    static getDerivedStateFromError(error: Error): Partial<State> {
      return { hasError: true, error };
    }

    componentDidCatch(error: Error, errorInfo: ErrorInfo) {
      console.error('[TiptapEditor] Editor crashed:', error, errorInfo);
    }

    handleRetry = () => {
      this.setState({ hasError: false, error: null });
    };

    render() {
      if (this.state.hasError) {
        return (
          <Box padding={4} background="danger100" hasRadius>
            <Typography variant="omega" textColor="danger700">
              The editor encountered an error and could not render.
            </Typography>
            <Box marginTop={2}>
              <Button variant="secondary" size="S" onClick={this.handleRetry}>
                Retry
              </Button>
            </Box>
          </Box>
        );
      }

      return this.props.children;
    }
  }
  ```

  The retry button calls `this.setState({ hasError: false, error: null })` which causes React to re-render, effectively remounting the children.
