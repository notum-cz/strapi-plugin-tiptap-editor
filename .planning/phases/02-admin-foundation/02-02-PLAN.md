  <action>
  Create `admin/src/components/EditorErrorBoundary.tsx` as a React class component.

  Per CONTEXT.md decisions:
  - Error boundary wraps `BaseTiptapInput` only (not the Field.Root/Label/Hint/Error)
  - Fallback UI: "Editor failed to load" message with a retry button that remounts the component
  - Use Strapi design-system `Alert` component (per user decision for consistent styling), or `Box` + `Typography` if Alert does not fit (Claude's discretion on exact component)
  - Log errors via `console.error` only
  - Retry button remounts by toggling a key (not by calling Tiptap API)

  Implementation:
  ```typescript
  import React, { Component, ErrorInfo, ReactNode } from 'react';
  import { Box, Typography, Button } from '@strapi/design-system';

  interface Props {
    children: ReactNode;
  }

  interface State {
    hasError: boolean;
    error: Error | null;
  }

  export class EditorErrorBoundary extends Component<Props, State> {
    state: State = { hasError: false, error: null };

    static getDerivedStateFromError(error: Error): Partial<State> {
      return { hasError: true, error };
    }

    componentDidCatch(error: Error, errorInfo: ErrorInfo) {
      console.error('[TiptapEditor] Editor crashed:', error, errorInfo);
    }

    handleRetry = () => {
      this.setState({ hasError: false, error: null });
    };

    render() {
      if (this.state.hasError) {
        return (
          <Box padding={4} background="danger100" hasRadius>
            <Typography variant="omega" textColor="danger700">
              The editor encountered an error and could not render.
            </Typography>
            <Box marginTop={2}>
              <Button variant="secondary" size="S" onClick={this.handleRetry}>
                Retry
              </Button>
            </Box>
          </Box>
        );
      }

      return this.props.children;
    }
  }
  ```

  The retry button calls `this.setState({ hasError: false, error: null })` which causes React to re-render, effectively remounting the children. This is the standard error boundary retry pattern.

  NOTE: This component is created in Phase 2 but only WIRED into RichTextInput in Phase 3. It just needs to exist and be exported correctly.
